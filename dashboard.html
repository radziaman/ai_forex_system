`${data.current_epoch}/${data.total_epochs}` : '--'; document.getElementById('currentEpoch').textContent = epochText; // Update models trained document.getElementById('modelsTrained').textContent = data.models_trained || 0; // Update completed pairs const completed = data.completed_pairs || []; document.getElementById('completedPairs').textContent = completed.length > 0 ? completed.join(', ') : 'None yet'; // Show/hide status messages const statusDiv = document.getElementById('trainingStatus'); const completeDiv = document.getElementById('trainingComplete'); const backtestDiv = document.getElementById('backtestResults'); if (data.training_complete) { statusDiv.style.display = 'none'; completeDiv.style.display = 'block'; progressBar.classList.remove('progress-bar-animated', 'progress-bar-striped'); // Show backtest results if (data.backtest_results && Object.keys(data.backtest_results).length > 0) { backtestDiv.style.display = 'block'; const tbody = document.getElementById('backtestTable'); tbody.innerHTML = Object.entries(data.backtest_results).map(([pair, r]) => ` ${pair.replace('=X', '')} ${r.total_return?.toFixed(2)}% ${r.win_rate?.toFixed(1)}% ${r.sharpe_ratio?.toFixed(2)} ${r.max_drawdown?.toFixed(2)}% `).join(''); } } else if (data.current_pair) { statusDiv.style.display = 'block'; completeDiv.style.display = 'none'; } }) .catch(err => console.error('Error fetching training status:', err)); } // Start training button handler function startTraining() { fetch('/api/training/start', { method: 'POST' }) .then(response => response.json()) .then(data => { alert('Training started! Progress will update automatically.'); // Start polling for updates setInterval(updateTrainingProgress, 2000); }) .catch(err => alert('Error starting training: ' + err)); } // Poll training status every 2 seconds setInterval(updateTrainingProgress, 2000); updateTrainingProgress(); // Initial call
